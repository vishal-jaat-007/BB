// Initialize
Razorpay razorpay = Razorpay();
razorpay.on(Razorpay.EVENT_PAYMENT_SUCCESS, handleSuccess);
razorpay.on(Razorpay.EVENT_PAYMENT_ERROR, handleError);

// Create order from backend
var orderResponse = await http.post(
  Uri.parse('$baseUrl/api/payment/create-order'),
  body: json.encode({
    'userId': userId,
    'amount': amount.toString(),
    ...
  }),
);

var orderData = json.decode(orderResponse.body);

// Open Razorpay Checkout
razorpay.open({
  'key': orderData['keyId'],
  'amount': orderData['amount'],  // Already in paise
  'name': 'Wallet Recharge',
  'order_id': orderData['orderId'],
  ...
});
